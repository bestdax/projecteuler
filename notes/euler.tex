\NewDocumentCommand{\contfrac}{m}
 {
  \seq_set_split:Nnn \l_tmpa_seq { , } { #1 }
  \seq_pop_left:NN \l_tmpa_seq \l_tmpa_tl % 弹出序列的第一个元素
  \l_tmpa_tl +
  \tl_set:Nn \l_tmpa_tl{}
  \contfrac_main: \l_tmpa_seq
 }

\cs_new_protected:Npn \contfrac_main: #1
 {
  \seq_map_inline:Nn #1
   {
     \tl_if_empty:NTF \l_tmpa_seq{}{
       \seq_pop_left:NN \l_tmpa_seq \l_tmpa_tl
       \cfrac{1}{\l_tmpa_tl +
         \tl_set:Nn \l_tmpa_tl{}
       \contfrac_main: \l_tmpa_seq}
     }
   }
 }
